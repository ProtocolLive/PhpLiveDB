Operators

Operators to compare values

 

Sintaxe

namespace ProtocolLive\PhpLiveDb;

enum Operators:string{
case Bigger = '>';
case BiggerEqual = '>=';
case Different = '<>';
case Equal = '=';
case In = ' in';
case IsNotNull = ' not null';
case Like = ' like ';
case NotIn = ' not in';
case Smaller = '<';
case SmallerEqual = '<=';
}
